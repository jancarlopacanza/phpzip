public function csvFileRead($filepath){
        
        if(isset($this->csvDataStore[$filepath])){
            return $this->csvDataStore[$filepath];
        }
        
        if(!file_exists($filepath)){
                return false;
                die();			
        }

        $skipfirstrow = false;
        $csvdata = array();
        if(($handle = fopen($filepath, 'r')) !== FALSE) {
                while (($data = fgetcsv($handle, 1000, ',', '"')) !== FALSE) {
                        if($skipfirstrow){
                                $csvdata[] = $data;
                        }

                        $skipfirstrow = true;
                }
                fclose($handle);
        }
        $this->csvDataStore[$filepath] = $csvdata;
        return $csvdata;
    }
